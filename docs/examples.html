Animator with Sprites

This example demonstrates how to use the Animator class to animate a sprite.

from engine.animation import Animator, AnimationCurve
from engine.sprite import Sprite
from engine.physics import Vec2
import skia

# Create an animator
animator = Animator()

# Create a sprite
image = skia.Image.MakeFromEncoded(skia.Data.MakeFromFileName("path/to/image.png"))
sprite = Sprite(image)
sprite.position = Vec2(100, 100)

# Create a named tween to animate the sprite's position
animator.to(
    "move_right",
    start=100.0,
    end=500.0,
    duration=2.0,
    curve=AnimationCurve.EASE_IN_OUT,
    on_complete=lambda: print("Animation complete")
)

# Update the animator and sprite position
def update(dt):
    animator.update(dt)
    tween = animator.get_tween("move_right")
    if tween:
        sprite.position.x = tween.value
    sprite.render(canvas, sprite.position)

Physics with Collision

This example demonstrates how to use the PhysicsWorld and CollisionWorld classes to simulate physics and handle collisions.

from engine.physics import PhysicsWorld, RigidBody, Vec2
from engine.collision import CollisionWorld

# Create a physics world with gravity
physics_world = PhysicsWorld(gravity=Vec2(0, 9.8))

# Create a collision world
collision_world = CollisionWorld()

# Create rigid bodies
body1 = RigidBody(position=Vec2(100, 100), velocity=Vec2(0, 0), mass=1.0)
body2 = RigidBody(position=Vec2(150, 150), velocity=Vec2(0, 0), mass=1.0)

# Add bodies to the physics world
physics_world.add_body(body1)
physics_world.add_body(body2)

# Add bodies to the collision world
collision_world.add_circle(body1, radius=20)
collision_world.add_circle(body2, radius=20)

# Update the physics world and check for collisions
def update(dt):
    physics_world.update(dt)
    collision_world.check_and_resolve()

Particle Effects

This example demonstrates how to use the ParticleSystem class to create particle effects.

from engine.particles import ParticleSystem
from engine.physics import Vec2
import skia

# Create a particle system
particle_system = ParticleSystem()

# Emit particles
particle_system.emit(
    pos=Vec2(100, 100),
    count=50,
    color=skia.ColorRED,
    speed_range=(50, 200),
    life_range=(0.3, 0.8),
    size_range=(2, 6),
    gravity=500.0
)

# Update and render the particle system
def update(dt):
    particle_system.update(dt)
    particle_system.render(canvas)

Post-Processing

This example demonstrates how to use the PostProcessSystem class to apply post-processing effects.

from engine.effects import PostProcessSystem

# Create a post-processing system
post_process = PostProcessSystem(engine)

# Trigger effects
post_process.trigger_shake(5.0)
post_process.trigger_glitch(1.0)

# Set the active shader
post_process.set_effect("glitch")

# Update the post-processing system
def update(dt):
    post_process.update(dt)

Asset Management

This example demonstrates how to use the AssetManager class to load and manage assets.

from engine.assets import AssetManager

# Get the asset manager instance
asset_manager = AssetManager.get()

# Load an image
image = asset_manager.load_image("path/to/image.png", "my_image")

# Load a spritesheet
spritesheet = asset_manager.load_spritesheet("path/to/spritesheet.png", frame_w=32, frame_h=32, key="my_spritesheet")

# Load a sound
asset_manager.load_sound("path/to/sound.wav", "my_sound")

# Play a sound
asset_manager.play_sound("my_sound", volume=1.0)

# Get a font
font = asset_manager.get_font("Arial", 14)
